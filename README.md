#### usage

```js
import useCSS from 'use-css'

function Oopsy(props) {
  const className = useCSS(`
    font-size: 21px;
    font-style: italic;
    color: ${props.color};
  `)

  return <div className={className}>because why not?</div>
}

render(<Oopsy color="green" />)
```
